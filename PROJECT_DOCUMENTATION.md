```python
#!/usr/bin/env python3
"""
Polymarket Dashboard - Main Pipeline
Orchestrates data collection, analysis, and aggregation
"""
import logging
import sys
from datetime import datetime
from pathlib import Path

# –ò–º–ø–æ—Ä—Ç –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–æ–≤
from scraper import fetch_polymarket_markets
from risk_calculator import enrich_with_risk_metrics
from news_fetcher import fetch_crypto_news
from news_analyzer import analyze_news_relevance
from aggregator import build_dashboard_data

# –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏—è
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('logs/pipeline.log'),
        logging.StreamHandler(sys.stdout)
    ]
)
logger = logging.getLogger(__name__)


def main():
    """Main pipeline execution"""
    start_time = datetime.now()
    logger.info("=" * 60)
    logger.info("Polymarket Dashboard Pipeline Started")
    logger.info("=" * 60)

    try:
        # Step 1: Fetch Polymarket markets
        logger.info("[1/5] Fetching Polymarket markets...")
        markets_data = fetch_polymarket_markets()
        logger.info(f"‚úì Fetched {len(markets_data)} markets")

        # Step 2: Calculate risk metrics
        logger.info("[2/5] Calculating risk metrics...")
        markets_with_metrics = enrich_with_risk_metrics(markets_data)
        logger.info(f"‚úì Enriched {len(markets_with_metrics)} markets with risk scores")

        # Step 3: Fetch crypto news
        logger.info("[3/5] Fetching crypto news (last 48h)...")
        news_data = fetch_crypto_news(hours=48)
        logger.info(f"‚úì Fetched {len(news_data)} news articles")

        # Step 4: Analyze news relevance
        logger.info("[4/5] Analyzing news relevance with Claude AI...")
        news_analysis = analyze_news_relevance(news_data, markets_with_metrics)
        logger.info(f"‚úì Analyzed {news_analysis['stats']['total_analyzed']} news-market pairs")
        logger.info(f"  - Claude AI: {news_analysis['stats']['claude_ai']}")
        logger.info(f"  - Keyword fallback: {news_analysis['stats']['keyword_fallback']}")

        # Step 5: Build final dashboard data
        logger.info("[5/5] Building dashboard data...")
        dashboard_data = build_dashboard_data(
            markets_with_metrics,
            news_analysis
        )

        # Save to frontend/data/
        output_path = Path("frontend/data/dashboard_data.json")
        output_path.parent.mkdir(parents=True, exist_ok=True)

        import json
        with open(output_path, 'w', encoding='utf-8') as f:
            json.dump(dashboard_data, f, indent=2, ensure_ascii=False)

        logger.info(f"‚úì Dashboard data saved to {output_path}")

        # Summary
        elapsed = (datetime.now() - start_time).total_seconds()
        logger.info("=" * 60)
        logger.info(f"Pipeline completed successfully in {elapsed:.1f}s")
        logger.info(f"Dashboard ready at: frontend/index.html")
        logger.info("=" * 60)

        return 0

    except Exception as e:
        logger.error(f"Pipeline failed: {str(e)}", exc_info=True)
        return 1


if __name__ == "__main__":
    sys.exit(main())
```

```python
    """
    Args:
        probability: float (0-1), –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –≤—ã–∏–≥—Ä—ã—à–∞
        price: float (0-1), —Ü–µ–Ω–∞ –≤ –¥–æ–ª–ª–∞—Ä–∞—Ö (0.18 = 18¬¢)
        volume: float, –æ–±—ä–µ–º —Ç–æ—Ä–≥–æ–≤ –≤ USD

    Returns:
        dict with score, multiplier, expected_value, kelly_fraction
    """

    # 1. Multiplier (–∏–∫—Å—ã)
    if price > 0:
        multiplier = 1 / price
    else:
        multiplier = 0

    # 2. Expected Value (–º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ)
    # EV = (prob √ó –≤—ã–∏–≥—Ä—ã—à) - ((1-prob) √ó –ø—Ä–æ–∏–≥—Ä—ã—à)
    # –≤—ã–∏–≥—Ä—ã—à = multiplier - 1 (–ø—Ä–∏–±—ã–ª—å)
    # –ø—Ä–æ–∏–≥—Ä—ã—à = 1 (—Å—Ç–∞–≤–∫–∞)
    ev = (probability * (multiplier - 1)) - (1 - probability)

    # 3. Kelly Criterion (–æ–ø—Ç–∏–º–∞–ª—å–Ω–∞—è –¥–æ–ª—è –∫–∞–ø–∏—Ç–∞–ª–∞)
    # f = (p √ó b - q) / b
    # –≥–¥–µ: p = probability, q = 1-p, b = multiplier-1
    if multiplier > 1:
        kelly_fraction = (probability * (multiplier - 1) - (1 - probability)) / (multiplier - 1)
        kelly_fraction = max(0, min(kelly_fraction, 1))  # clamp [0, 1]
    else:
        kelly_fraction = 0

    # 4. Liquidity weight (–¥–ª—è –Ω–∞–¥–µ–∂–Ω–æ—Å—Ç–∏)
    # –ò—Å–ø–æ–ª—å–∑—É–µ–º logarithmic scale, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –Ω—É–ª–µ–≤—ã—Ö –≤–µ—Å–æ–≤
    # $10k = 0.5, $100k = 0.75, $1M = 1.0
    import math
    if volume > 0:
        # log scale: log10(volume) normalized to [0, 1]
        # min=$10k (log=4), max=$10M (log=7)
        volume_weight = min(max((math.log10(volume) - 4) / 3, 0.1), 1.0)
    else:
        volume_weight = 0.1  # –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–π –≤–µ—Å –¥–ª—è –Ω—É–ª–µ–≤–æ–≥–æ volume

    # 5. Final score (–∫–æ–º–±–∏–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–π)
    # –£—á–∏—Ç—ã–≤–∞–µ–º EV, Kelly –∏ –ª–∏–∫–≤–∏–¥–Ω–æ—Å—Ç—å
    score = ev * kelly_fraction * volume_weight

    return {
        'score': score,
        'multiplier': round(multiplier, 2),
        'expected_value': round(ev, 4),
        'kelly_fraction': round(kelly_fraction, 4),
        'volume_weight': round(volume_weight, 4)
    }
```

```python
def keyword_fallback_analysis(news_title, market_question, market_outcomes):
    """
    –ü—Ä–æ—Å—Ç–æ–π keyword matching –∫–∞–∫ backup
    """
    news_lower = news_title.lower()
    question_lower = market_question.lower()

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è outcomes –≤ –∑–∞–≥–æ–ª–æ–≤–∫–µ –Ω–æ–≤–æ—Å—Ç–∏
    matching_outcomes = []
    for outcome in market_outcomes:
        if outcome.lower() in news_lower:
            matching_outcomes.append(outcome)

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –∫–ª—é—á–µ–≤—ã—Ö —Å–ª–æ–≤ –∏–∑ –≤–æ–ø—Ä–æ—Å–∞
    question_keywords = set(question_lower.split()) - {'the', 'a', 'an', 'will', 'be', 'is'}
    news_keywords = set(news_lower.split())
    overlap = len(question_keywords & news_keywords)

    # –†–∞—Å—á–µ—Ç –ø—Ä–æ—Å—Ç–æ–≥–æ relevance score
    relevance_score = 0
    if matching_outcomes:
        relevance_score += 5
    relevance_score += min(overlap * 0.5, 5)

    return {
        "relevance_score": min(relevance_score, 10),
        "affects_outcomes": matching_outcomes,
        "impact_direction": "neutral",
        "confidence": "low",
        "reasoning": f"Keyword match: {len(matching_outcomes)} outcomes found",
        "analysis_method": "keyword_fallback"
    }
```

```python
def build_closing_soon_widget(markets_data):
    """
    –†—ã–Ω–∫–∏, –∑–∞–∫—Ä—ã–≤–∞—é—â–∏–µ—Å—è –≤ —Ç–µ—á–µ–Ω–∏–µ 24 —á–∞—Å–æ–≤

    opportunities = []

    for market in markets_data:
        end_date = datetime.fromisoformat(market['end_date'].replace('Z', '+00:00'))
        hours_until_close = (end_date - now).total_seconds() / 3600

        # –§–∏–ª—å—Ç—Ä: –∑–∞–∫—Ä—ã—Ç–∏–µ –≤ —Ç–µ—á–µ–Ω–∏–µ 24h

                    reward_factor = (mult - 1) * 100  # –ø—Ä–∏–±—ã–ª—å –≤ %
                    closing_score = time_factor * prob * reward_factor

                    opportunities.append({

                        'market_url': market['url'],
                        'outcome_name': outcome['name'],
                        'probability': prob,
                        'multiplier': mult,

                        'volume': outcome['volume']

                        reward_factor = (mult - 1) * 100
                        closing_score = time_factor * prob * reward_factor

                        sorted_opps.append({
                            'market_title': market['question'],
                            'market_url': market['url'],
                            'outcome_name': outcome['name'],
                            'probability': prob,

        sorted_opps = sorted(sorted_opps, key=lambda x: x['closing_score'], reverse=True)
```

```python
        market = next((m for m in markets_data if m['id'] == market_id), None)

        if market:
            top_news.append({
                'news_title': best_news['news_title'],
                'market_title': market['question'],
                'market_url': market['url'],
                'relevance_score': best_news['relevance_score'],
                'affects_outcomes': best_news['affects_outcomes'],
                'impact_direction': best_news['impact_direction'],
                'confidence': best_news['confidence'],
                'reasoning': best_news['reasoning']
            })

    # –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ relevance_score
    return sorted(top_news, key=lambda x: x['relevance_score'], reverse=True)[:10]
```

### Risk-Reward Score
```
Score = EV √ó Kelly √ó Liquidity

–≥–¥–µ:
EV = (P √ó (M - 1)) - (1 - P)
Kelly = (P √ó M - 1) / (M - 1)
Liquidity = min(Volume / $1M, 1.0)

P = Probability (0-1)
M = Multiplier
```

### Closing Score
```
Closing Score = Time Factor √ó Probability √ó Reward Factor

–≥–¥–µ:
Time Factor = 1 / (Hours Until Close + 1)
Reward Factor = (Multiplier - 1) √ó 100
```

### Relevance Score (Claude AI)
- Semantic analysis –Ω–æ–≤–æ—Å—Ç–∏ vs –æ–ø–∏—Å–∞–Ω–∏–µ —Ä—ã–Ω–∫–∞
- Keyword matching (outcomes)
- Contextual understanding
- Output: 0-10 scale
